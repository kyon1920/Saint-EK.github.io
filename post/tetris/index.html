俄罗斯方块

2019.06.22

经典小游戏：俄罗斯方块



项目名称：俄罗斯方块
开发工具：VS2019
开发语言：C/C++、windowsAPI
项目介绍：运行程序时，将听到一首背景音乐，按下回车键开始游戏，并播放开始提示音，然后进入游戏，播放背景音乐，游戏种存在不同形态的6种随机方块，按下左右方向键改变方向，上键改变方块形态，下键加速下落，游戏规则同经典游戏的俄罗斯方块。

具体实现：
1、游戏图形：整个为500╳659的窗体，游戏区域由300╳600窗体区域构成，为什么游戏区域高度与窗体高度不符，因为窗体存在非客户区，非客户区占有一定的高度。

2、实现逻辑：整个游戏中，分为设备坐标和逻辑坐标两类，逻辑坐标决定方块形状和位置，而设备坐标决定方块显示的位置，逻辑坐标决定图形-由产生随机数函数产生1-6的随机数，将1-6的数字分别绑定方块的6种不同形态，再将方块映射到对应的设备坐标上显示出来，并将前面的实现都放入定时器中，没过设定的时间，方块便下落一个正方形方块宽带的距离，然后进行消息处理，接收键盘消息，如果键盘点击了左方向键，方块左移一个单元，键盘点击了右方向键，方块右移一个单元，键盘点击了上方向键，方块变形-变形通过改变逻辑坐标的值来实现，键盘点击了下方向键，改变定时器的值，使方块加速下落。最后进行边界判断和满行判断，边界判断通过设备坐标来实现，通过方块值是否到达最左边或者最右边来实现是否可以继续移动，满行判断-正在移动的方块被设定的值为1、下落完成的方块的值被设定为2，当判断设备坐标中出现整行值为2的话便实现了满行，并进行消除，将分数+1，最后将消除行以上并且处于值为2的方块全部向下落下一格，实现消行操作。

//void OnCreate()函数
void OnCreate()
{
    srand((unsigned int)time(NULL));

    CopySqareToBack();

//设置游戏兼容性DC
void OnPaint(HDC hDc)
{
    HDC hMemDC = CreateCompatibleDC(hDc);
    HBITMAP hBitmap = CreateCompatibleBitmap(hDc, 500, 600);
    SelectObject(hMemDC, hBitmap);
    PaintSqare(hMemDC);
    ShowSqare2(hMemDC);
    //传递
    ShowScore(hDc);
    BitBlt(hDc, 0, 0, 300, 600, hMemDC, 0, 0, SRCCOPY);
    DeleteObject(hBitmap);
    DeleteDC(hMemDC);
}

//显示方块
void PaintSqare(HDC hDc)
{
    //画游戏方框
    Rectangle(hDc, 0, 0, 300, 600);
    HPEN hPen, hOldPen;
    HBRUSH hBrush, hOldBrush;
    //画方块
    hPen = CreatePen(BS_SOLID, 1, RGB(0, 255, 0));
    hOldPen = (HPEN)SelectObject(hDc, hPen);
    hBrush = CreateSolidBrush(RGB(0, 0, 255));
    hOldBrush = (HBRUSH)SelectObject(hDc, hBrush);
    for (int i = 0; i < 20; ++i)
    {
        for (int j = 0; j < 10; ++j)
        {
            if (1 == g_arrBackGround[i][j])
            {
                Rectangle(hDc, j * 30, i * 30, j * 30 + 30, i * 30 + 30);
            }
        }
    }
    SelectObject(hDc, hOldPen);
    SelectObject(hDc, hOldBrush);
}

//产生随机块
int CreateRandenSqare()
{
    int nIndex = rand() % 7;
    switch (nIndex)
    {
    case 0:
        g_arrSqare[0][0] = 1, g_arrSqare[0][1] = 1, g_arrSqare[0][2] = 0, g_arrSqare[0][3] = 0;
        g_arrSqare[1][0] = 0, g_arrSqare[1][1] = 1, g_arrSqare[1][2] = 1, g_arrSqare[1][3] = 0;
        g_nLine = 0;
        g_nList = 3;

    case 1:

        g_arrSqare[1][0] = 1, g_arrSqare[1][1] = 1, g_arrSqare[1][2] = 0, g_arrSqare[1][3] = 0;
        g_nLine = 0;
        g_nList = 3;
        break;
    case 2:
        g_arrSqare[0][0] = 1, g_arrSqare[0][1] = 0, g_arrSqare[0][2] = 0, g_arrSqare[0][3] = 0;
        g_arrSqare[1][0] = 1, g_arrSqare[1][1] = 1, g_arrSqare[1][2] = 1, g_arrSqare[1][3] = 0;
        g_nLine = 0;
        g_nList = 3;
        break;
    case 3:
        g_arrSqare[0][0] = 0, g_arrSqare[0][1] = 0, g_arrSqare[0][2] = 1, g_arrSqare[0][3] = 0;
        g_arrSqare[1][0] = 1, g_arrSqare[1][1] = 1, g_arrSqare[1][2] = 1, g_arrSqare[1][3] = 0;
        g_nLine = 0;
        g_nList = 3;
        break;
    case 4:
        g_arrSqare[0][0] = 0, g_arrSqare[0][1] = 1, g_arrSqare[0][2] = 0, g_arrSqare[0][3] = 0;
        g_arrSqare[1][0] = 1, g_arrSqare[1][1] = 1, g_arrSqare[1][2] = 1, g_arrSqare[1][3] = 0;
        g_nLine = 0;
        g_nList = 3;
        break;
    case 5:
        g_arrSqare[0][0] = 0, g_arrSqare[0][1] = 1, g_arrSqare[0][2] = 1, g_arrSqare[0][3] = 0;
        g_arrSqare[1][0] = 0, g_arrSqare[1][1] = 1, g_arrSqare[1][2] = 1, g_arrSqare[1][3] = 0;
        g_nLine = 0;
        g_nList = 4;
        break;
    case 6:
        g_arrSqare[0][0] = 1, g_arrSqare[0][1] = 1, g_arrSqare[0][2] = 1, g_arrSqare[0][3] = 1;
        g_arrSqare[1][0] = 0, g_arrSqare[1][1] = 0, g_arrSqare[1][2] = 0, g_arrSqare[1][3] = 0;
        g_nLine = 0;
        g_nList = 4;
        break;
    }
    g_nSqareID = nIndex;
    return nIndex;
}

//定时器响应函数
void OnTimer(HWND hWnd)
{
    HDC hdc = GetDC(hWnd);
    if (1 == CanSqareDown() && 1 == CanSqareDown2())
    {
        SqareDown();
        g_nLine++;
    }
    else
    {
        //1 to 2
        Change1To2();
        //消除
        DestroyOneLineSqare();
        //产生随机块
        //游戏结束
        if (0 == GameOver())
        {
            //结束应用程序
            KillTimer(hWnd, DEF_TIME1);
            return ;
        }
        CreateRandenSqare();
        CopySqareToBack();
    }
    //显示方块
    //PaintSqare(hdc);
    OnPaint(hdc);
    ReleaseDC(hWnd, hdc);
}

void OnLeft(HWND hWnd)
{

        HDC hdc = GetDC(hWnd);
        //方块左移
        SqareLeft();
        g_nList--;
        //显示方块
        OnPaint(hdc);
        ReleaseDC(hWnd, hdc);
    }
}

void OnRight(HWND hWnd)
{
    if (1 == CanSqareRight() && 1 == CanSqareRight2())
    {
        HDC hdc = GetDC(hWnd);
        //方块右移
        SqareRight();
        g_nList++;
        //显示方块
        OnPaint(hdc);
        ReleaseDC(hWnd, hdc);

}

//方块下落
void SqareDown()
{
    for (int i = 19; i >= 0; i--)
    {
        for (int j = 0; j < 10; ++j)
        {
            if (1 == g_arrBackGround[i][j])
            {
                g_arrBackGround[i + 1][j] = g_arrBackGround[i][j];
                g_arrBackGround[i][j] = 0;
            }
        }
    }
}

//按下键加速移动
void OnDown(HWND hWnd)
{
    OnTimer(hWnd);
}

                g_arrBackGround[i][j] = 0;
            }
        }

void ChangeSqare()

//长条变形
void ChangeLineSqare()
{
    if (1 == g_arrBackGround[g_nLine][g_nList - 1])
    {
        //清零
        g_arrBackGround[g_nLine][g_nList - 1] = 0;
        g_arrBackGround[g_nLine][g_nList + 1] = 0;
        g_arrBackGround[g_nLine][g_nList + 2] = 0;

        if (2 == g_arrBackGround[g_nLine + 1][g_nList])
        {
            //元素赋值
            g_arrBackGround[g_nLine - 1][g_nList] = 1;
            g_arrBackGround[g_nLine - 2][g_nList] = 1;
            g_arrBackGround[g_nLine - 3][g_nList] = 1;
        }
        else if(2 == g_arrBackGround[g_nLine + 2][g_nList])
        {
            //元素赋值
            g_arrBackGround[g_nLine + 1][g_nList] = 1;
            g_arrBackGround[g_nLine - 1][g_nList] = 1;
            g_arrBackGround[g_nLine - 2][g_nList] = 1;
        }
        else
        {
            //元素赋值
            g_arrBackGround[g_nLine - 1][g_nList] = 1;
            g_arrBackGround[g_nLine + 1][g_nList] = 1;
            g_arrBackGround[g_nLine + 2][g_nList] = 1;
        }
        
    }
    else
    {
        //同上
        g_arrBackGround[g_nLine - 1][g_nList] = 0;
        g_arrBackGround[g_nLine + 1][g_nList] = 0;
        g_arrBackGround[g_nLine + 2][g_nList] = 0;

        if (g_nList == 9 || g_nList - 3 >= 0 && 2 == g_arrBackGround[g_nLine][g_nList + 1])
        {
            g_arrBackGround[g_nLine][g_nList - 1] = 1;
            g_arrBackGround[g_nLine][g_nList - 2] = 1;
            g_arrBackGround[g_nLine][g_nList - 3] = 1;
            g_nList -= 2;
        }
        else if (g_nList == 8 || g_nList - 2 >= 0 && 2 == g_arrBackGround[g_nLine][g_nList + 2])
        {
            g_arrBackGround[g_nLine][g_nList - 2] = 1;
            g_arrBackGround[g_nLine][g_nList - 1] = 1;
            g_arrBackGround[g_nLine][g_nList + 1] = 1;
            g_nList -= 1;
        }
        else if (0 == g_nList || g_nList + 3 <= 19 && 2 == g_arrBackGround[g_nLine][g_nList - 1])
        {
            g_arrBackGround[g_nLine][g_nList + 1] = 1;
            g_arrBackGround[g_nLine][g_nList + 2] = 1;
            g_arrBackGround[g_nLine][g_nList + 3] = 1;
            g_nList += 1;
        }
        else
        {
            g_arrBackGround[g_nLine][g_nList - 1] = 1;
            g_arrBackGround[g_nLine][g_nList + 1] = 1;
            g_arrBackGround[g_nLine][g_nList + 2] = 1;
        }
        
    }
}

//判断长条能否变形
int CanLineSqareChange()
{
    int i = 1, j = 1;
    for (i = 1; i < 4; ++i)
    {
        if (2 == g_arrBackGround[g_nLine][g_nList + i] || g_nList + i > 9)
        {
            break;
        }
    }
    for (j = 1; j < 4; ++j)
    {
        if (2 == g_arrBackGround[g_nLine][g_nList - j] || g_nList - j < 0)
        {
            break;
        }
    }
    if ((i - 1 + j - 1) < 3)
    {
        return 0;
    }
    return 1;
}

#define MAX_LOADSTRING 100

BOOL                InitInstance(HINSTANCE, int);

int APIENTRY wWinMain(In HINSTANCE hInstance,
                     _Inopt HINSTANCE hPrevInstance,
                     In LPWSTR    lpCmdLine,
                     In int       nCmdShow)
{
    UNREFERENCED_PARAMETER(hPrevInstance);
    UNREFERENCED_PARAMETER(lpCmdLine);

    // TODO: 在此处放置代码。

    // 初始化全局字符串
    LoadStringW(hInstance, IDS_APP_TITLE, szTitle, MAX_LOADSTRING);
    LoadStringW(hInstance, IDC_MY, szWindowClass, MAX_LOADSTRING);
    MyRegisterClass(hInstance);

    // 执行应用程序初始化:
    if (!InitInstance (hInstance, nCmdShow))
    {
        return FALSE;
    }

    HACCEL hAccelTable = LoadAccelerators(hInstance, MAKEINTRESOURCE(IDC_MY));

    MSG msg;

    // 主消息循环:
    while (GetMessage(&msg, nullptr, 0, 0))
    {
        if (!TranslateAccelerator(msg.hwnd, hAccelTable, &msg))
        {
            TranslateMessage(&msg);
            DispatchMessage(&msg);
        }
    }

    return (int) msg.wParam;
}

//  函数: MyRegisterClass()
//
//  目标: 注册窗口类。

ATOM MyRegisterClass(HINSTANCE hInstance)
{
    WNDCLASSEXW wcex;

    wcex.cbSize = sizeof(WNDCLASSEX);

    wcex.style          = CS_HREDRAW | CS_VREDRAW;
    wcex.lpfnWndProc    = WndProc;
    wcex.cbClsExtra     = 0;
    wcex.cbWndExtra     = 0;
    wcex.hInstance      = hInstance;
    wcex.hIcon          = LoadIcon(hInstance, MAKEINTRESOURCE(IDI_MY));
    wcex.hCursor        = LoadCursor(nullptr, IDC_ARROW);
    wcex.hbrBackground  = (HBRUSH)COLOR_APPWORKSPACE;
    wcex.lpszMenuName   = MAKEINTRESOURCEW(IDC_MY);
    wcex.lpszClassName  = szWindowClass;
    wcex.hIconSm        = LoadIcon(wcex.hInstance, MAKEINTRESOURCE(IDI_SMALL));

    return RegisterClassExW(&wcex);
}

//
//   函数: InitInstance(HINSTANCE, int)
//

BOOL InitInstance(HINSTANCE hInstance, int nCmdShow)

//
//  函数: WndProc(HWND, UINT, WPARAM, LPARAM)
//
//  目标: 处理主窗口的消息。
//
//  WM_COMMAND  - 处理应用程序菜单
//  WM_PAINT    - 绘制主窗口
//  WM_DESTROY  - 发送退出消息并返回
//
//
LRESULT CALLBACK WndProc(HWND hWnd, UINT message, WPARAM wParam, LPARAM lParam)
{
    HDC hDc;

    switch (message)
    {
    case WM_CREATE:
        PlayMusic();
        OnCreate();
        break;
    case WM_COMMAND:
        {
            int wmId = LOWORD(wParam);
            // 分析菜单选择:
            switch (wmId)
            {
            case IDM_ABOUT:
                DialogBox(hInst, MAKEINTRESOURCE(IDD_ABOUTBOX), hWnd, About);
                break;
            case IDM_EXIT:
                DestroyWindow(hWnd);
                break;
            default:
                return DefWindowProc(hWnd, message, wParam, lParam);
            }
        }
        break;
    case WM_TIMER:
        OnTimer(hWnd);
        break;
    case WM_KEYDOWN:
        switch (wParam)
        {
        case VK_RETURN:
            OnReturn(hWnd);
            PlaySound(TEXT("开场白.wav"), NULL, SND_FILENAME | SND_ASYNC);
            Sleep(2000);
            PlaySound(TEXT("安和桥.wav"), NULL, SND_FILENAME | SND_ASYNC);
            break;
        case VK_LEFT:
            OnLeft(hWnd);
            break;
        case VK_RIGHT:
            OnRight(hWnd);
            break;
        case VK_UP:
            OnChangeSqare(hWnd);
            break;
        case VK_DOWN:
            OnDown(hWnd);
            break;
        }
        break;
    case WM_PAINT:
        {
            PAINTSTRUCT ps;
            hDc = BeginPaint(hWnd, &ps);
            // TODO: 在此处添加使用 hdc 的任何绘图代码...
            TextOut(hDc, 310, 50, TEXT("当前游戏得分:"), 7);
            TextOut(hDc, 310, 260, TEXT("游戏规则:"), 5);
            TextOut(hDc, 310, 280, TEXT("按回车键开始游戏"), 8);
            TextOut(hDc, 310, 300, TEXT("由方向键控制"), 6);
            TextOut(hDc, 310, 320, TEXT("↑ 变形"), 4);
            TextOut(hDc, 310, 340, TEXT("← 左移"), 4);
            TextOut(hDc, 310, 360, TEXT("→ 右移"), 4);
            TextOut(hDc, 310, 380, TEXT("↓ 加速下落"), 6);
            TextOut(hDc, 310, 400, TEXT("填满一行得1分，递增"), 10);
            OnPaint(hDc);
            EndPaint(hWnd, &ps);
        }
        break;
    case WM_DESTROY:
        KillTimer(hWnd, DEF_TIME1);
        PostQuitMessage(0);
        break;
    default:
        return DefWindowProc(hWnd, message, wParam, lParam);
    }
    return 0;
}

// “关于”框的消息处理程序。
INT_PTR CALLBACK About(HWND hDlg, UINT message, WPARAM wParam, LPARAM lParam)
{
    UNREFERENCED_PARAMETER(lParam);
    switch (message)
    {
    case WM_INITDIALOG:
        return (INT_PTR)TRUE;

    case WM_COMMAND:
        if (LOWORD(wParam) == IDOK || LOWORD(wParam) == IDCANCEL)
        {
            EndDialog(hDlg, LOWORD(wParam));
            return (INT_PTR)TRUE;
        }
        break;
    }
    return (INT_PTR)FALSE;
}
