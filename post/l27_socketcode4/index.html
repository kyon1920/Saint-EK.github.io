int threadpool_add(threadpool_t *pool, void*(*function)(void *arg), void *arg)
{

    /* ==ä¸ºçœŸï¼Œé˜Ÿåˆ—å·²ç»æ»¡ï¼Œ è°ƒwaité˜»å¡ */
    while ((pool->queue_size == pool->queue_max_size) && (!pool->shutdown)) {
        pthread_cond_wait(&(pool->queue_not_full), &(pool->lock));
    }
    if (pool->shutdown) {
        pthread_mutex_unlock(&(pool->lock));
    }

    /* æ¸…ç©º å·¥ä½œçº¿ç¨‹ è°ƒç”¨çš„å›è°ƒå‡½æ•° çš„å‚æ•°arg */
    if (pool->task_queue[pool->queue_rear].arg != NULL) {
        free(pool->task_queue[pool->queue_rear].arg);
        pool->task_queue[pool->queue_rear].arg = NULL;
    }
    /*æ·»åŠ ä»»åŠ¡åˆ°ä»»åŠ¡é˜Ÿåˆ—é‡Œ*/
    pool->task_queue[pool->queue_rear].function = function;
    pool->task_queue[pool->queue_rear].arg = arg;
    pool->queue_rear = (pool->queue_rear + 1) % pool->queue_max_size;       /* é˜Ÿå°¾æŒ‡é’ˆç§»åŠ¨, æ¨¡æ‹Ÿç¯å½¢ */
    pool->queue_size++;

    /*æ·»åŠ å®Œä»»åŠ¡åï¼Œé˜Ÿåˆ—ä¸ä¸ºç©ºï¼Œå”¤é†’çº¿ç¨‹æ± ä¸­ ç­‰å¾…å¤„ç†ä»»åŠ¡çš„çº¿ç¨‹*/
    pthread_cond_signal(&(pool->queue_not_empty));
    pthread_mutex_unlock(&(pool->lock));

    return 0;
}

void *threadpool_thread(void *threadpool)
{
    threadpool_t *pool = (threadpool_t *)threadpool;
    threadpool_task_t task;

    while (true) {
        /* Lock must be taken to wait on conditional variable */
        /*åˆšåˆ›å»ºå‡ºçº¿ç¨‹ï¼Œç­‰å¾…ä»»åŠ¡é˜Ÿåˆ—é‡Œæœ‰ä»»åŠ¡ï¼Œå¦åˆ™é˜»å¡ç­‰å¾…ä»»åŠ¡é˜Ÿåˆ—é‡Œæœ‰ä»»åŠ¡åå†å”¤é†’æ¥æ”¶ä»»åŠ¡*/

        /*queue_size == 0 è¯´æ˜æ²¡æœ‰ä»»åŠ¡ï¼Œè°ƒ wait é˜»å¡åœ¨æ¡ä»¶å˜é‡ä¸Š, è‹¥æœ‰ä»»åŠ¡ï¼Œè·³è¿‡è¯¥while*/
        while ((pool->queue_size == 0) && (!pool->shutdown)) {  
            printf("thread 0x%x is waiting\n", (unsigned int)pthread_self());
            pthread_cond_wait(&(pool->queue_not_empty), &(pool->lock));

            /*æ¸…é™¤æŒ‡å®šæ•°ç›®çš„ç©ºé—²çº¿ç¨‹ï¼Œå¦‚æœè¦ç»“æŸçš„çº¿ç¨‹ä¸ªæ•°å¤§äº0ï¼Œç»“æŸçº¿ç¨‹*/
            if (pool->wait_exit_thr_num > 0) {
                pool->wait_exit_thr_num--;

                /*å¦‚æœçº¿ç¨‹æ± é‡Œçº¿ç¨‹ä¸ªæ•°å¤§äºæœ€å°å€¼æ—¶å¯ä»¥ç»“æŸå½“å‰çº¿ç¨‹*/
                if (pool->live_thr_num > pool->min_thr_num) {
                    printf("thread 0x%x is exiting\n", (unsigned int)pthread_self());
                    pool->live_thr_num--;
                    pthread_mutex_unlock(&(pool->lock));
                    pthread_exit(NULL);
                }
            }
        }

        /*å¦‚æœæŒ‡å®šäº†trueï¼Œè¦å…³é—­çº¿ç¨‹æ± é‡Œçš„æ¯ä¸ªçº¿ç¨‹ï¼Œè‡ªè¡Œé€€å‡ºå¤„ç†*/

            pthread_mutex_unlock(&(pool->lock));
            printf("thread 0x%x is exiting\n", (unsigned int)pthread_self());
            pthread_exit(NULL);     /* çº¿ç¨‹è‡ªè¡Œç»“æŸ */
        }

        /*ä»ä»»åŠ¡é˜Ÿåˆ—é‡Œè·å–ä»»åŠ¡, æ˜¯ä¸€ä¸ªå‡ºé˜Ÿæ“ä½œ*/

        task.arg = pool->task_queue[pool->queue_front].arg;

        pool->queue_size--;

        pthread_mutex_unlock(&(pool->lock));

        /*æ‰§è¡Œä»»åŠ¡*/ 
        printf("thread 0x%x start working\n", (unsigned int)pthread_self());
        pthread_mutex_lock(&(pool->thread_counter));                            /*å¿™çŠ¶æ€çº¿ç¨‹æ•°å˜é‡ç*/
        pool->busy_thr_num++;                                                   /*å¿™çŠ¶æ€çº¿ç¨‹æ•°+1*/
        pthread_mutex_unlock(&(pool->thread_counter));
        (*(task.function))(task.arg);                                           /*æ‰§è¡Œå›è°ƒå‡½æ•°ä»»åŠ¡*/
        //task.function(task.arg);                                              /*æ‰§è¡Œå›è°ƒå‡½æ•°ä»»åŠ¡*/

        /*ä»»åŠ¡ç»“æŸå¤„ç†*/

        pthread_mutex_unlock(&(pool->thread_counter));
    }

                    && pool->live_thr_num < pool->max_thr_num; i++) {
                if (pool->threads[i] == 0 || !is_thread_alive(pool->threads[i])) {
                    pthread_create(&(pool->threads[i]), NULL, threadpool_thread, (void *)pool);
                    add++;
                    pool->live_thr_num++;
                }
            }

            pthread_mutex_unlock(&(pool->lock));
        }

        /* é”€æ¯å¤šä½™çš„ç©ºé—²çº¿ç¨‹ ç®—æ³•ï¼šå¿™çº¿ç¨‹X2 å°äº å­˜æ´»çš„çº¿ç¨‹æ•° ä¸” å­˜æ´»çš„çº¿ç¨‹æ•° å¤§äº æœ€å°çº¿ç¨‹æ•°æ—¶*/
        if ((busy_thr_num * 2) < live_thr_num  &&  live_thr_num > pool->min_thr_num) {

            /* ä¸€æ¬¡é”€æ¯DEFAULT_THREADä¸ªçº¿ç¨‹, éš¨æ©Ÿ10å€‹å³å¯ */
            pthread_mutex_lock(&(pool->lock));
            pool->wait_exit_thr_num = DEFAULT_THREAD_VARY;      /* è¦é”€æ¯çš„çº¿ç¨‹æ•° è®¾ç½®ä¸º10 */
            pthread_mutex_unlock(&(pool->lock));

            for (i = 0; i < DEFAULT_THREAD_VARY; i++) {
                /* é€šçŸ¥å¤„åœ¨ç©ºé—²çŠ¶æ€çš„çº¿ç¨‹, ä»–ä»¬ä¼šè‡ªè¡Œç»ˆæ­¢*/
                pthread_cond_signal(&(pool->queue_not_empty));
            }
        }
    }

    return NULL;
}

int threadpool_destroy(threadpool_t *pool)
{
    int i;
    if (pool == NULL) {
        return -1;
    }
    pool->shutdown = true;

    /*å…ˆé”€æ¯ç®¡ç†çº¿ç¨‹*/
    pthread_join(pool->adjust_tid, NULL);

    for (i = 0; i < pool->live_thr_num; i++) {
        /*é€šçŸ¥æ‰€æœ‰çš„ç©ºé—²çº¿ç¨‹*/
        pthread_cond_broadcast(&(pool->queue_not_empty));
    }
    for (i = 0; i < pool->live_thr_num; i++) {
        pthread_join(pool->threads[i], NULL);
    }
    threadpool_free(pool);

    return 0;
}

int threadpool_free(threadpool_t *pool)
{
    if (pool == NULL) {
        return -1;
    }

    if (pool->task_queue) {
        free(pool->task_queue);
    }
    if (pool->threads) {
        free(pool->threads);
        pthread_mutex_lock(&(pool->lock));
        pthread_mutex_destroy(&(pool->lock));
        pthread_mutex_lock(&(pool->thread_counter));
        pthread_mutex_destroy(&(pool->thread_counter));
        pthread_cond_destroy(&(pool->queue_not_empty));
        pthread_cond_destroy(&(pool->queue_not_full));
    }
    free(pool);
    pool = NULL;

    return 0;
}

int threadpool_all_threadnum(threadpool_t *pool)
{
    int all_threadnum = -1;
    pthread_mutex_lock(&(pool->lock));
    all_threadnum = pool->live_thr_num;
    pthread_mutex_unlock(&(pool->lock));
    return all_threadnum;
}

{
    int busy_threadnum = -1;
    pthread_mutex_lock(&(pool->thread_counter));
    busy_threadnum = pool->busy_thr_num;
    pthread_mutex_unlock(&(pool->thread_counter));
    return busy_threadnum;
}

{

    if (kill_rc == ESRCH) {

}
