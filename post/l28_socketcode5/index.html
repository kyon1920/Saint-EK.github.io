TCP总结
TCP：面向连接的可靠的数据报传递 ---完全弥补
😄 优点：
     数据包传输稳定，依赖于TCP的回执机制，丢包概率大概维护在 97/1000
     传输速率稳定，所经过路由一般是固定的
     流量稳定，因为TCP内部自带滑动窗口机制
🙁 缺点：
     效率低，需要进行三次握手，发送完成之后还需要等待客户端回执，效率大概为UDP的65%
     速度慢
🏆 适用场景：
     适用于传输大文件，重要文件传输

相比较于TCP
UDP：无连接的不可靠报文传递 ---完全不弥补
😄 优点:
     效率高、速度快
🙁 缺点：
     不稳定(数据、速率、流量)，路由不固定先发的数据包可能后到达，后发的数据包也许会先到达
     数据丢失概率高
🏆 适用场景：
     对实时性要求较高的场合，例如：视频会议、视频电话、广播
在实际当中使用TCP、UDP就不是那么绝对，例如腾讯使用的是TCP+UDP，在传输文件时用UDP并在应用层自定义协议弥补UDP丢包，而在中小型公司没有实现自定义协议的精力和能力，一般都是采用TCP。

UDP服务器概念
◾ 传输层主要应用的协议模型有两种，一种是TCP协议，另外一种则是UDP协议。TCP协议在网络通信中占主导地位，绝大多数的网络通信借助TCP协议完成数据传输。但UDP也是网络通信中不可或缺的重要通信手段。
◾ 相较于TCP而言，UDP通信的形式更像是发短信。不需要在数据传输之前建立、维护连接。只专心获取数据就好。省去了三次握手的过程，通信速度可以大大提高，但与之伴随的通信的稳定性和正确率便得不到保证。因此，我们称UDP为“无连接的不可靠报文传递”。
◾ 那么与我们熟知的TCP相比，UDP有哪些优点和不足呢？由于无需创建连接，所以UDP开销较小，数据传输速度快，实时性较强。多用于对实时性要求较高的通信场合，如视频会议、电话会议等。但随之也伴随着数据传输不可靠，传输数据的正确率、传输顺序和流量都得不到控制和保证。所以，通常情况下，使用UDP协议进行数据传输，为保证数据的正确性，我们需要在应用层添加辅助校验协议来弥补UDP的不足，以达到数据可靠传输的目的。
◾ 与TCP类似的，UDP也有可能出现缓冲区被填满后，再接收数据时丢包的现象。由于它没有TCP滑动窗口的机制，通常采用如下两种方法解决：
     服务器应用层设计流量控制，控制发送数据速度
     借助setsockopt函数改变接收缓冲区大小，如：

#include <sys/types.h>
#include <ctype.h>
#include <netinet/in.h>

using namespace std;
const int MAXLINE = 80;
const int SERV_PORT = 6006;

int main()
{
        struct sockaddr_in saddr, caddr;
        socklen_t clen;
        int sockfd;
        int ret, n;
        char buf[MAXLINE], str[INET_ADDRSTRLEN];

        sockfd = socket(AF_INET, SOCK_DGRAM, 0);
        if (sockfd == -1)
        {
                cerr << "Create socket error!" << endl;
                exit(1);
        }
        cout << "Create socket success!" << endl;

        bzero(&saddr, sizeof(saddr));
        saddr.sin_family = AF_INET;
        saddr.sin_addr.s_addr = INADDR_ANY;
        saddr.sin_port = htons(SERV_PORT);

        ret = bind(sockfd, (struct sockaddr *)&saddr, sizeof(saddr));
        if (ret == -1)
        {
                cerr << "Socket bind failed!" << endl;
                exit(1);
        }
        cout << "Socket bind success!" << endl;

        cout << "Accept connections ..." << endl;

        while(1)
        {
                clen = sizeof(caddr);
                n = recvfrom(sockfd, buf, MAXLINE, 0, 
                                (struct sockaddr *)&caddr, &clen);
                if (n == -1)
                {
                        cerr << "recvfrom error!" << endl;
                        exit(1);
                }

                cout << "Received from IP " <<inet_ntop(AF_INET, 
                                &caddr.sin_addr, str, sizeof(str))<< 
                        " and the PORT is " << ntohs(caddr.sin_port) << endl;

                for (int i = 0; i < n; ++i)
                {
                        buf[i] = toupper(buf[i]);
                }

                n = sendto(sockfd, buf, n, 0, (struct sockaddr *)&caddr,
                                sizeof(caddr));
                if (n == -1)
                {
                        cerr << "Server sendto error!" << endl;
                        exit(1);
                }
        }
        close(sockfd);
        return 0;
}

#include <stdlib.h>
#include <stdio.h>
#include <unistd.h>
#include <string.h>
#include <strings.h>
#include <arpa/inet.h>
#include <sys/socket.h>
#include <sys/types.h>

using namespace std;
const int MAXLINE = 80;
const int SERV_PORT = 6006;

int main()
{
        struct sockaddr_in saddr;
        int n, sockfd;
        char buf[MAXLINE];

        sockfd = socket(AF_INET, SOCK_DGRAM, 0);

        bzero(&saddr, sizeof(saddr));
        saddr.sin_family = AF_INET;
        inet_pton(AF_INET, "127.0.0.1", &saddr.sin_addr.s_addr);
        saddr.sin_port = htons(SERV_PORT);

        while(fgets(buf, MAXLINE, stdin) != NULL)
        {
                n = sendto(sockfd, buf, strlen(buf), 0, 
                                (struct sockaddr *)&saddr, sizeof(saddr));
                if (n == -1)
                {
                        cerr << "Sendto error!" << endl;
                        exit(1);
                }

                n = recvfrom(sockfd, buf, MAXLINE, 0, NULL, 0);
                if (n == -1)
                {
                        cerr << "Recvfrom error!" << endl;
                        exit(1);
                }

                write(STDOUT_FILENO, buf, n);
        }
        close (sockfd);
        return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <arpa/inet.h>
#include <net/if.h>

using namespace std;
const int SERVER_PORT = 8000;
const int CLIENT_PORT = 9000;
const int MAXLINE = 1500;

const char* BROADCAST_IP = "192.168.113.255";

int main()
{
        int sockfd, ret;
        struct sockaddr_in saddr, caddr;
        char buf[MAXLINE];

        /*构造用于UDP通信的套接字*/
        sockfd = socket(AF_INET, SOCK_DGRAM, 0);
        if (sockfd == -1)
        {
                cerr << "Create socket error!" << endl;
                exit(1);
        }
        cout << "Create socket success!" << endl;

        bzero(&saddr, sizeof(saddr));
        saddr.sin_family = AF_INET;
        saddr.sin_addr.s_addr = htonl(INADDR_ANY);
        saddr.sin_port = htons(SERVER_PORT);

        ret = bind(sockfd, (struct sockaddr *)&saddr, sizeof(saddr));
        if (-1 == ret)
        {
                cerr << "Socket bind failed!" << endl;
                exit(1);
        }
        cout << "Socket bind success!" << endl;

        int opt = 1;
        setsockopt(sockfd, SOL_SOCKET, SO_BROADCAST, 
                        &opt, sizeof(opt));

        /* 构造client地址：IP + 端口 192.168.113.255 */
        bzero(&caddr, sizeof(caddr));
        caddr.sin_family = AF_INET;
        inet_pton(AF_INET, BROADCAST_IP, &caddr.sin_addr.s_addr);
        caddr.sin_port = htons(CLIENT_PORT);

        int i = 0;
        while(1)
        {
                sprintf(buf, "Drink %d glasses of water\n", i++);
                // fgets(buf, sizeof(buf), stdin);
                sendto(sockfd, buf, strlen(buf), 0, 
                                (struct sockaddr *)&caddr, sizeof(caddr));
                sleep(1);
        }

        close(sockfd);
        return 0;
}

多播(组播)
🤨 组播组可以是永久的也可以是临时的。组播组地址中，有一部分由官方分配的，称为永久组播组。永久组播组保持不变的是它的ip地址，组中的成员构成可以发生变化。永久组播组中成员的数量都可以是任意的，甚至可以为零。那些没有保留下来供永久组播组使用的ip组播地址，可以被临时组播组利用。
     224.0.0.0～224.0.0.255 为预留的组播地址（永久组地址），地址224.0.0.0保留不做分配，其它地址供路由协议使用；
     224.0.1.0～224.0.1.255 是公用组播地址，可以用于Internet；欲使用需申请；
     224.0.2.0～238.255.255.255 为用户可用的组播地址（临时组地址），全网范围内有效；
     239.0.0.0～239.255.255.255 为本地管理组播地址，仅在特定的本地范围内有效；
🤨 可使用ip ad命令查看网卡编号，如：
     itcast\$ ip ad
       1: lo: mtu 65536 qdisc noqueue state UNKNOWN group default
          link/loopback 00:00:00:00:00:00 brd 00:00:00:00:00:00
          inet 127.0.0.1/8 scope host lo
              valid_lft forever preferred_lft forever
          inet6 ::1/128 scope host
              valid_lft forever preferred_lft forever
       2: eth0: mtu 1500 qdisc pfifo_fast state DOWN group default qlen 1000
          link/ether 00:0c:29:0a:c4:f4 brd ff:ff:ff:ff:ff:ff
          inet6 fe80::20c:29ff:fe0a:c4f4/64 scope link
              valid_lft forever preferred_lft forever
     <>内表示网卡支持的服务
     mtu：最大传输单元
     link/ether：网卡编号
🤨 if_nametoindex 命令 / if_nametoindex() 函数 可以根据网卡名，获取网卡序号
🤨 补充：ifconfig eth0(网卡名称) up/down 命令启动/关闭网卡
     int setsockopt(int sockfd, int level, int optname, const void *optval, socklen_t optlen);
     server端：
         设置获取组播权限
         setsockopt(sockfd, IPPROTO_IP, IP_MULTICAST_IF, &group, sizeof(group));
     client端：
         将本客户端加入组播
         setsockopt(confd, IPPROTO_IP, IP_ADD_MEMBERSHIP, &group, sizeof(group));
     setsockopt的作用总结：
         0、端口复用
         1、设置缓冲区大小
         2、开放广播权限
         3、开放组播权限
         4、加入组播组

#include <string.h>
#include <arpa/inet.h>
#include <net/if.h>

#define SERVER_PORT 8000
#define CLIENT_PORT 9000
#define MAXLINE 1500

#define GROUP "239.0.0.2"

int main(void)
{
    int sockfd;
    struct sockaddr_in serveraddr, clientaddr;
    char buf[MAXLINE] = "itcast\n";
    struct ip_mreqn group;

    sockfd = socket(AF_INET, SOCK_DGRAM, 0);  /* 构造用于UDP通信的套接字 */
    
    bzero(&serveraddr, sizeof(serveraddr));
    serveraddr.sin_family = AF_INET;                        /* IPv4 */
    /* 本地任意IP INADDR_ANY = 0 */
    serveraddr.sin_addr.s_addr = htonl(INADDR_ANY);
    serveraddr.sin_port = htons(SERVER_PORT);

    bind(sockfd, (struct sockaddr *)&serveraddr, sizeof(serveraddr));

    inet_pton(AF_INET, GROUP, &group.imr_multiaddr);    /* 设置组地址 */
    inet_pton(AF_INET, "0.0.0.0", &group.imr_address);      /* 本地任意IP */
    /* 给出网卡名,转换为对应编号: eth0 --> 编号  命令:ip ad */
    group.imr_ifindex = if_nametoindex("eth0");

    /* 组播权限 */
    setsockopt(sockfd, IPPROTO_IP, IP_MULTICAST_IF, &group, sizeof(group));

    bzero(&clientaddr, sizeof(clientaddr));                 /* 构造 client 地址 IP+端口 */
    clientaddr.sin_family = AF_INET;
    inet_pton(AF_INET, GROUP, &clientaddr.sin_addr.s_addr); /* IPv4  239.0.0.2+9000 */
    clientaddr.sin_port = htons(CLIENT_PORT);

    int i = 0;
    while (1) {
        sprintf(buf, "itcast %d\n", i++);
        //fgets(buf, sizeof(buf), stdin);
        sendto(sockfd, buf, strlen(buf), 0, (struct sockaddr *)&clientaddr, sizeof(clientaddr));
        sleep(1);
    }

    close(sockfd);
    return 0;
}

#include <stdio.h>
#include <unistd.h>
#include <string.h>
#include <arpa/inet.h>
#include <net/if.h>

#define SERVER_PORT 8000
#define CLIENT_PORT 9000

#define GROUP "239.0.0.2"

int main(int argc, char *argv[])
{
    struct sockaddr_in localaddr;
    int confd;
    ssize_t len;
    char buf[BUFSIZ];

    struct ip_mreqn group;                                                  /* 组播结构体 */

    confd = socket(AF_INET, SOCK_DGRAM, 0);

    bzero(&localaddr, sizeof(localaddr));                            /* 初始化 */
    localaddr.sin_family = AF_INET;
    inet_pton(AF_INET, "0.0.0.0" , &localaddr.sin_addr.s_addr);
    localaddr.sin_port = htons(CLIENT_PORT);

    bind(confd, (struct sockaddr *)&localaddr, sizeof(localaddr));

    inet_pton(AF_INET, GROUP, &group.imr_multiaddr);     /* 设置组地址 */
    inet_pton(AF_INET, "0.0.0.0", &group.imr_address);      /* 使用本地任意IP添加到组播组 */
    group.imr_ifindex = if_nametoindex("eth0");                 /* 通过网卡名-->编号 ip ad */
    /* 设置client 加入多播组 */
    setsockopt(confd, IPPROTO_IP, IP_ADD_MEMBERSHIP, &group, sizeof(group));

    while (1) {
        len = recvfrom(confd, buf, sizeof(buf), 0, NULL, 0);
        write(STDOUT_FILENO, buf, len);
    }

    close(confd);
    return 0;
}
