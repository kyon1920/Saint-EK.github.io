#include <iostream>
#include <stdlib.h>
#include <unistd.h>
#include <pthread.h>
using namespace std;
pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;
pthread_cond_t cond = PTHREAD_COND_INITIALIZER;
struct has_product
{
        int node;
        struct has_product *next;
};
struct has_product *head = NULL;
struct has_product *mp;
void *Producer(void *arg)
{
        struct has_product *tail;
        for (;;)
        {
                mp = new struct has_product;
                mp -> node = rand()%1000 + 1;
                mp -> next = NULL;
                pthread_mutex_lock(&mutex);
                tail = head;
                if (head == NULL)
                {
                        head = mp;
                }
                else
                {
                        while(tail -> next != NULL)
                        {
                                tail = tail -> next;
                        }
                        tail -> next = mp;
                }
                cout << "Producer: " << head -> node << endl;
                pthread_mutex_unlock(&mutex);
                pthread_cond_signal(&cond);
                sleep(rand()%3 + 1);
        }
}

void *Consumer(void *arg)
{
        struct has_product *pie;
        for(;;)
        {
                pthread_mutex_lock(&mutex);
                while(head == NULL)
                {
                        pthread_cond_wait(&cond, &mutex);
                }
                pie = head;
                head = pie -> next;
                pthread_mutex_unlock(&mutex);
                cout << "Cunsumer: " << mp -> node << endl;
                free(pie);
                sleep(rand()%10 + 10);
        }
}

int main()
{
        srand(time(NULL));
        int ret;
        pthread_t proid, conid[5];
        ret = pthread_create(&proid, NULL, Producer, NULL);
        if (0 == ret)
        {
                cout << "Prodecer create success!" << endl;
        }
        else
        {
                cout << "Producer create failed!" << endl;
                exit(1);
        }
        for(int i = 0; i < 5; ++i)
        {
                ret = pthread_create(&(conid[i]), NULL, Consumer, NULL);
                if (0 == ret)
                {
                        cout << "The " << i + 1 << "th consumer create success!" << endl;
                }
                else
                {
                        cout << "The " << i + 1 << "th producer create failed!" << endl;

                }
        }

        for (int i = 0; i < 5; ++i)
        {
                pthread_join(conid[i], NULL);
        }*/
        while(1);
        return 0;
}

{
        int node;
        struct has_product *next;
};
struct has_product *head = NULL;
struct has_product *mp;
void *Producer(void *arg)
{
        struct has_product *tail;
        for (;;)
        {
                mp = new struct has_product;
                mp -> node = rand()%1000 + 1;
                mp -> next = NULL;
                sem_trywait(&sem);
                tail = head;
                if (head == NULL)
                {
                        head = mp;
                }
                else
                {
                        while(tail -> next != NULL)
                        {
                                tail = tail -> next;
                        }
                        tail -> next = mp;
                }
                cout << "Producer: " << head -> node << endl;
                sem_post(&sem);
                sleep(rand()%3 + 1);
        }
}

void *Consumer(void *arg)
{
        struct has_product *pie;
        for(;;)
        {
                while(head == NULL)
                {
                }
                sem_trywait(&sem);
                pie = head;
                head = pie -> next;
                sem_post(&sem);
                cout << "Cunsumer: " << mp -> node << endl;
                free(pie);
                sleep(rand()%10 + 10);
        }
}

{
        sem_init(&sem, 0, 4);

哲学家用餐模型-多线程版
🙃 选用互斥锁mutex，如创建5个， pthread_mutex_t tid[5];
模型抽象：
         5个哲学家 → 5个线程； 5支筷子 → 5把互斥锁
         5个哲学家使用相同的逻辑，可通用一个线程主函数，void *Philfun(void *arg)，使用参数来表示线程编号：long i = (long)arg;
         哲学家线程根据编号知道自己是第几个哲学家，而后选定锁，锁住，吃饭，否则哲学家thinking。
         5支筷子，在逻辑上形成环： 0 1 2 3 4 分别对应5个哲学家：
     振荡：如果每个人都攥着自己左手的锁，尝试去拿右手锁，拿不到则将锁释放，过会儿五个人又同时再攥着左手锁尝试拿右手锁，依然拿不到，如此往复形成另外一种极端死锁的现象 —— 振荡。
     避免振荡现象：只需5个人中，任意一个人，拿锁的方向与其他人相逆即可(如：哲学家0，原来：1、0 现在：0、1)。
     主线程(main)中，初始化5把锁，销毁5把锁，创建5个线程（并将i传递给线程主函数），回收5个线程。
🙃 避免死锁的方法：
     0、当得不到所有所需资源时，放弃已经获得的资源，等待。
     1、保证资源的获取顺序，要求每个线程获取资源的顺序一致。如：A获取顺序1、2、3；B顺序应也是1、2、3。若B为3、2、1则易出现死锁现象。

#include <unistd.h>
#include <pthread.h>
#include <stdlib.h>
using namespace std;
pthread_mutex_t tid[5];
int flg = 0;
void *Philfun(void *arg)
{
        int ret;
        long i = (long)arg;
        if (i == 0)
        {
                cout << "0 try take first chopsticks..." << endl;
                while(0 != pthread_mutex_trylock(&(tid[0])))
                {
                }
                cout << "0 take first chopsticks success!" << endl;
                while(0 != pthread_mutex_trylock(&(tid[1])))
                {
                }
                cout << "0 Dining complete!" << endl;
                pthread_mutex_unlock(&(tid[0]));
                pthread_mutex_unlock(&(tid[1]));
                flg++;
        }
        else if (i == 4)
        {
                cout << "4 try take first chopsticks..." << endl;
                while(0 != pthread_mutex_trylock(&(tid[0])))
                {
                }
                cout <<"4 take first chopsticks success!" << endl;
                while(0 != pthread_mutex_trylock(&(tid[4])))
                {
                }
                cout <<"4 Dining complete!" << endl;
                pthread_mutex_unlock(&(tid[0]));
                pthread_mutex_unlock(&(tid[4]));
                flg++;
        }
        else
        {
                cout << i << " try take first chopsticks..." << endl;
                while(0 != pthread_mutex_trylock(&(tid[i + 1])))
                {
                }
                cout << i << " take first chopsticks success!" << endl;
                while(0 != pthread_mutex_trylock(&(tid[i])))
                {
                }
                cout << i << " Dining complete!" << endl;
                pthread_mutex_unlock(&(tid[i]));
                pthread_mutex_unlock(&(tid[i + 1]));
                flg++;
        }
}

int main()
{
        int ret;
        pthread_t m[5];
        for (int i = 0; i < 5; ++i)
        {
                ret = pthread_mutex_init(&(tid[i]), NULL);
                if (ret == 0)
                {
                        cout << "Init the " << i + 1 << "th lock success!" << endl;
                }
                else
                {
                        cout << "Init the " << i + 1 << "th lock failed!" << endl;
                        exit(1);
                }

        }
        for (int i = 0; i < 5; ++i)
        {
                ret = pthread_create(&(m[i]), NULL, Philfun, (void *)i);
                if (ret == 0)
                {
                        cout << "Create the " << i + 1 << "th pthread success!" << endl;
                }
                else
                {
                        cout << "Create the " << i + 1 << "th pthread failed!" << endl;
                        exit(1);
                }
        }
        while(flg != 5)
        {
        }
        for (int i = 0; i < 5; ++i)
        {
                pthread_mutex_destroy(&(tid[i]));
                pthread_cancel(m[i]);
                pthread_join(m[i], &ret);
        }
        return 0;
}


哲学家用餐模型-多进程版
🙃 相较于多线程需注意问题：
     需注意如何共享信号量 (注意：坚决不能使用全局变量 sem_t sem[5])
     实现：
     main函数中：
         循环 sem_init(&sem[i], 0, 1); 将信号量初值设为1，信号量变为互斥锁。
         循环 sem_destroy(&sem[i]);
         循环 创建 5 个子进程。 if(i < 5) 中完成子进程的代码逻辑。
         循环 回收 5 个子进程。
     注意：直接将sem_t sem[5]放在全局位置，试图用于子进程间共享是错误的！应将其定义放置与mmap共享映射区中或者main中：
         sem_t *sem = mmap(NULL, sizeof(sem_t) * 5, PROT_READ|PROT_WRITE, MAP_SHARED|MAP_ANON， -1， 0);
         使用方式：将sem当成数组首地址看待，与使用数组s[5]没有差异
