pthread_mutex_t mutex = PTHREAD_MUTEX_INITIALIZER;
pthread_cond_t cond = PTHREAD_COND_INITIALIZER;
struct has_product
{
        int node;
        struct has_product *next;
};
struct has_product *head = NULL;
struct has_product *mp;
void *Producer(void *arg)
{
        struct has_product *tail;
        for (;;)
        {
                mp = new struct has_product;
                mp -> node = rand()%1000 + 1;
                mp -> next = NULL;
                pthread_mutex_lock(&mutex);
                tail = head;
                if (head == NULL)
                {
                        head = mp;
                }
                else
                {
                        while(tail -> next != NULL)
                        {
                                tail = tail -> next;
                        }
                        tail -> next = mp;
                }
                cout << "Producer: " << head -> node << endl;
                pthread_mutex_unlock(&mutex);
                pthread_cond_signal(&cond);
                sleep(rand()%3 + 1);
        }
}

void *Consumer(void *arg)
{
        struct has_product *pie;
        for(;;)
        {
                pthread_mutex_lock(&mutex);

                {
                        pthread_cond_wait(&cond, &mutex);
                }
                pie = head;
                head = pie -> next;
                pthread_mutex_unlock(&mutex);
                cout << "Cunsumer: " << mp -> node << endl;
                free(pie);
                sleep(rand()%10 + 10);
        }
}
