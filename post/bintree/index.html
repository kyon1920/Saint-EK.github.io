#include <iostream>
#include <stdlib.h>
#include <stack>
#include <queue>
using namespace std;

typedef struct node
{
        char data;
        struct node *lchild, *rchild;
}BiNode, *BiTree;

// 初始化一棵二叉树
void InitBiTree(BiTree bTree)
{
        bTree = NULL;
}

// 销毁一棵二叉树
void DestroyBiTree(BiTree bTree)
{
        if (bTree == NULL)
        {
                return ;
        }
        DestroyBiTree(bTree -> lchild);
        DestroyBiTree(bTree -> rchild);
        cout << "已销毁节点：" << bTree -> data << endl;
        delete bTree;
        bTree = NULL;
}

// 判断二叉树是否为空
bool BiTreeEmpty(const BiTree bTree)
{
        if (bTree == NULL)
        {
                return true;
        }
        else
        {
                return false;
        }
}

// 查看根节点信息
void Root(const BiTree bTree)
{
        cout << "根节点：" << bTree -> data << endl;
}

// 创建一棵二叉树：非递归
BiTree CreateBiTree()
{
        string str;
        cin >> str;
        char *ch;
        ch  = new char[str.length()];
        for (int i = 0; i < str.length(); ++i)
        {
                ch[i] = str.at(i);
        }
        int flag = 1, num = 0;
        stack<BiTree> sta;

        if (ch[num] == '*')
        {
                return NULL;
        }

        BiTree tmp;
        BiTree root = new BiNode();
        root -> data = ch[num];
        root -> lchild = NULL;
        root -> rchild = NULL;
        num++;

        sta.push(root);

        while(num < str.length())
        {
                BiTree pNew = NULL;

                if (flag == 1)
                {
                        if (ch[num] == '*')
                        {
                                flag = 2;
                        }
                        else
                        {
                                pNew = new BiNode();
                                pNew -> data = ch[num];
                                pNew -> lchild = NULL;
                                pNew -> rchild = NULL;

                                tmp = sta.top();
                                tmp -> lchild = pNew;
                                sta.push(pNew);
                                flag = 1;
                        }
                }
                else if (flag == 2)
                {
                        if (ch[num] == '*')
                        {
                                flag = 3;
                        }
                        else
                        {
                                pNew = new BiNode();
                                pNew -> data = ch[num];
                                pNew -> lchild = NULL;
                                pNew -> rchild = NULL;

                                tmp = sta.top();
                                tmp -> rchild = pNew;
                                sta.push(pNew);
                                flag = 1;
                        }
                }
                else
                {
                        tmp = sta.top();
                        sta.pop();
                        while (sta.size() > 0 &&
                            sta.top() -> rchild == tmp)
                        {
                                tmp = sta.top();
                                sta.pop();
                        }
                        flag = 2;
                        --num;
                }
                ++num;
        }
        return root;
}

// 创建一个二叉树：递归
void CreateBiTree(BiTree& bTree)
{
        char tdata;
        cin >> tdata;
        if (tdata == '*')
        {
                bTree = NULL;
        }
        else
        {
                bTree = new BiNode();
                bTree -> data = tdata;
                bTree -> lchild = NULL;
                bTree -> rchild = NULL;
                CreateBiTree(bTree -> lchild);
                CreateBiTree(bTree -> rchild);
        }
}

// 先序遍历：非递归
void PreOrderTraverse(const BiTree bTree)
{
        BiTree p;

        stack<BiTree> sta;
        sta.push(bTree);

        while(sta.empty() != true)
        {
                while((p = sta.top()) != NULL)
                {
                        cout << p -> data << " ";
                        sta.push(p -> lchild);
                }
                sta.pop();
                if (sta.empty() != true)
                {
                        p = sta.top();
                        sta.pop();
                        sta.push(p -> rchild);
                }
        }
        cout << endl;
}

// 先序遍历：递归
void PriorOrderTraverse(const BiTree bTree)
{
        if (bTree)
        {
                cout << bTree -> data << " ";
                PriorOrderTraverse(bTree -> lchild);
                PriorOrderTraverse(bTree -> rchild);
        }
}

// 中序遍历：非递归
void InOrderTraverse(BiTree bTree)
{
        stack<BiTree> sta;
        sta.push(bTree);

        BiTree p;

        while(sta.empty() != true)
        {
                while((p = sta.top()) != NULL)
                {
                        sta.push(p -> lchild);
                }
                sta.pop();
                if (sta.empty() != true)
                {
                        p = sta.top();
                        sta.pop();
                        cout << p -> data << " ";
                        sta.push(p -> rchild);
                }
        }
        cout << endl;
}

// 中序遍历：递归
void MiddleOrderTraverse(BiTree T)
{
        if (T)
        {
                MiddleOrderTraverse(T -> lchild);
                cout << T -> data << " ";
                MiddleOrderTraverse(T -> rchild);
        }
}

// 后序遍历：非递归
void LastOrderTraverse(const BiTree bTree)
{
        BiTree p, r;

        stack<BiTree> sta;
        sta.push(bTree);

        while(sta.empty() != true)
        {
                while((p = sta.top()) != NULL)
                {
                        sta.push(p -> lchild);
                }
                sta.pop();
                if (sta.empty() != true)
                {
                        p = sta.top();
                        while (p -> rchild == NULL ||
                                p -> rchild == r)
                        {
                                cout << p -> data << " ";
                                r = p;
                                sta.pop();
                                if (sta.empty() != true)
                                {
                                        p = sta.top();
                                }
                                else
                                {
                                        break;
                                }
                        }
                        if (sta.empty() != true)
                        {
                                sta.push(p -> rchild);
                        }
                }
        }
        cout << endl;
}

// 后序遍历：递归
void PostOrderTraverse(const BiTree bTree)
{
        if (bTree)
        {
                PostOrderTraverse(bTree -> lchild);
                PostOrderTraverse(bTree -> rchild);
                cout << bTree -> data << " ";
        }
}

//层序遍历
void LevelOrderTraverse(const BiTree bTree)
{
        queue<BiTree> que;
        que.push(bTree);
        while(que.empty() != true)
        {
                cout << que.front() -> data << " ";
                if (que.front() -> lchild != NULL)
                {
                        que.push(que.front() -> lchild);
                }
                if (que.front() -> rchild != NULL)
                {
                        que.push(que.front() -> rchild);
                }
                que.pop();
        }
        cout << endl;
}

// 返回节点的值
char Value(const BiTree bTree)
{
        char c;
        cout << "请输入你需要查询的节点：";
        cin >> c;

        stack<BiTree> sta;
        BiTree p;
        sta.push(bTree);
        p = sta.top();

        while (sta.empty() != true)
        {
                while ((p = sta.top()) != NULL)
                {
                        sta.push(p -> lchild);
                }
                sta.pop();
                if (sta.empty() != true)
                {
                        p = sta.top();
                        sta.pop();
                        if (p -> data == c)
                        {
                                return p -> data;
                        }
                        sta.push(p -> rchild);
                }
        }
        return '*';
}

// 返回双亲节点
BiTree Parent(BiTree bTree)
{
        char c;
        cout << "请输入你需要查询的节点：";
        cin >> c;

        stack<BiTree> sta;
        BiTree p;
        sta.push(bTree);
        p = sta.top();

        while (sta.empty() != true)
        {
                while ((p = sta.top()) != NULL)
                {
                        sta.push(p -> lchild);
                }
                sta.pop();
                if (sta.empty() != true)
                {
                        p = sta.top();
                        if (p -> lchild != NULL)
                        {
                                if (p -> lchild -> data == c)
                                {
                                        return p;
                                }
                        }
                        if (p -> rchild != NULL)
                        {
                                if (p -> rchild -> data == c)
                                {
                                        return p;
                                }
                        }
                        sta.pop();
                        sta.push(p -> rchild);
                }
        }
        return NULL;
}

int main()
{
        char c = '\0';
        BiTree bTree;
        BiTree ret = NULL;

        // 初始化二叉树
        cout << "======初始化二叉树======" << endl;
        InitBiTree(bTree);
        cout << "初始化完成." << endl;

        cout << "======创建二叉树======" << endl;
        cout << "递归创建还是非递归？D/F: ";
        cin >> c;
        if (c == 'F' || c == 'f')
        {
                bTree = CreateBiTree();
        }
        else if (c == 'D' || c == 'd')
        {
                CreateBiTree(bTree);
        }

        // 判断二叉树是否为空
        if (BiTreeEmpty(bTree) == true)
        {
                cout << "创建了一棵空二叉树." << endl;
                exit(1);
        }

        cout << "======查看根节点======" << endl;
        Root(bTree);

        cout << "======先序遍历======" << endl;
        cout << "非递归遍历：";
        PreOrderTraverse(bTree);
        cout << "递归遍历：";
        PriorOrderTraverse(bTree);
        cout << endl;
        cout << "======中序遍历======" << endl;
        cout << "非递归遍历: ";
        InOrderTraverse(bTree);
        cout << "递归遍历: ";
        MiddleOrderTraverse(bTree);
        cout << endl;
        cout << "======后序遍历======" << endl;
        cout << "非递归实现：";
        LastOrderTraverse(bTree);
        cout << "递归实现：";
        PostOrderTraverse(bTree);
        cout << endl;
        cout << "======层序遍历======" << endl;
        LevelOrderTraverse(bTree);

        cout << "======查询节点数据======" << endl;
        c = Value(bTree);
        cout << c << endl;
        if (c == '*')
        {
                cout << "该节点不存在." << endl;
        }
        else
        {
                cout << "节点存在，且数据为 " << c << endl;
        }

        cout << "======查询节点的双亲节点======" << endl;
        ret = Parent(bTree);
        if (ret == NULL)
        {
                cout << "该节点无双亲节点." << endl;
        }
        else
        {
                cout << "双亲节点为 " << ret -> data << endl;
        }

        // 销毁二叉树
        cout << "======销毁二叉树======" << endl;
        DestroyBiTree(bTree);
        cout << "销毁完成." << endl;

        return 0;
}
