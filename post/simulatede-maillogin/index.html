模拟E-Mail登录程序

2018.12.16

利用C/C++语言模拟E-Mail的登录


项目名称：模拟E-Mail登录程序
开发工具：VS2019
开发语言：C/C++
项目介绍：本次项目实现了模拟E-Mail的登录，并实现了userName的检查，如果符合基本邮箱格式，即可输入密码，否则重新输入userName，再输入密码，用户名和密码都验证通过后，将会弹出登录进度条，进度条到达100%以后提示用户登录成功，然后通过system函数打开电脑内置记事本程序。

具体实现：
1、界面实现：全部通过C语言system函数实现，设置标题、背景颜色和窗体大小，具体代码如下：

3、用户名格式判断和密码验证：用户名的判断是通过库中的正则表达式实现，相关邮箱格式则来自网络(我暂时还没有学习正则表达式)，如果符合正则表达式对应格式则通过用户名的输入，进入密码的输入，否则通过system函数实现清屏，重新显示界面并重新输入用户名，密码的判断十分简单，只要和录入的密码一致时则可进行登录，否则提示用户”密码或者用户名错误“并进行清屏重头开始；
4、登录进度条的实现：；利用循环和清屏函数实现，循环是加多'■'字符的个数和百分数的大小，每一次循环进行一次清屏和显示，直到进度条达到100%；
5、登录成功，打开记事本。

#include <iostream>
#include <string>
#include <regex>    //正则库
#include <cstdio>
#include "conio.h"  //_getch() and _putch()
#include <Windows.h>

using namespace std;
/*******************************/
//  调整窗口大小
//  system函数+dos指令
//  标题 title 标题名
//  颜色 color f0
//  大小 mode con cols = 50 lines = 10
void SetWindowStyle()
{
    system("title 邮箱登陆");
    system("color f0");
    system("mode con cols=50 lines=10");
}

/*******************************/
//  检查用户名
//  利用正则表达式来检查用户名
//  如果是正确的邮箱格式返回true，否则返回false
bool CheckEmail(char* username)
{
    regex object("^[A-Za-z\d]+([-_.][A-Za-z\d]+)*@([A-Za-z\d]+[-.])+[A-Za-z\d]{2,4}\$");
    bool result = regex_match(username, object);
    return result;
}

/*******************************/
//给用户给予登陆提示
void SetTips()
{
    cout << "\n\n\n\t\t  Welcome!" << endl;
    Sleep(3000);
    system("cls");
}

/*******************************/
//  实现进度条
void proc()
{
    string str("■");
    for (int i = 0; i < 20; i++)
    {
        system("cls");
        cout << str << endl;
        cout << (i + 1) * 5 << "%" << endl;
        str += "■";
        Sleep(200);
    }
    cout << "登陆成功！" << endl;
}

void InputUserEmail(char* username, char* password)
{
    cout << "\n\n\tUsername: ";
    cin >> username;
    /*******************************/
    //检查用户名格式
    //不正确提示用户重新输入
    while (!CheckEmail(username))
    {
        cout << "用户名格式不正确，请重新输入！" << endl;
        Sleep(1000);
        system("cls");
        cout << "\n\n\tUsername: ";
        cin >> username;
    }
    cout << "\tPassword: ";
    /*******************************/
    //密码不可见，字符串当作字符处理
    //每次按键输入一个*就可以 然后把所有按键保存到password中
    char key;
    int i = 0;
    while ((key = _getch()) != '\r')
    {
        if (i < 8)
        {
            password[i++] = key;
            _putch('*');
        }
        else
        {
            cout << "\nPassword too long，请重新输入！" << endl;
            Sleep(1000);
            system("cls");
            cout << "\n\n\tUsername: " << username << endl;
            cout << "\tPassword: ";
            i = 0;
        }
    }
    /*******************************/
    //字符串结束标准
    password[i] = '\0';
    cout << endl;
}

int main()
{
    /*******************************/
    //用户名+密码
    char username[20] = "";
    char password[9] = "";
    SetWindowStyle();
    SetTips();
    InputUserEmail(username, password);
    while (strcmp(username, "137989531@qq.com") || strcmp(password, "12345"))
    {
        cout << "用户名或者密码错误，请检查后重新输入！\n";
        Sleep(1000);
        system("cls");
        InputUserEmail(username, password);
    }
    //弹出进度条
    proc();
    //打开记事本
    system("notepad");
    return 0;
}



存在缺陷：只能实现语言内设置用户的登录，无法增加用户和进行注册，这项技术应该由数据库来完成。
